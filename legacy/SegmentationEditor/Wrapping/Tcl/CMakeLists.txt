# create the WSTITK executable
ADD_EXECUTABLE(WSTITK tkAppInit.cxx)
INSTALL_TARGETS(/bin WSTITK)

TARGET_LINK_LIBRARIES (WSTITK
  vtkImagingTCL
  vtkGraphicsTCL
  vtkIOTCL
  vtkFilteringTCL
  vtkCommonTCL
)

IF (VTK_USE_RENDERING)
  TARGET_LINK_LIBRARIES(WSTITK vtkRenderingTCL)
ENDIF (VTK_USE_RENDERING)

IF (VTK_USE_PATENTED)
  TARGET_LINK_LIBRARIES(WSTITK vtkPatentedTCL)
ENDIF (VTK_USE_PATENTED)

IF (VTK_USE_HYBRID)
  TARGET_LINK_LIBRARIES(WSTITK vtkHybridTCL)
ENDIF (VTK_USE_HYBRID)

IF (VTK_USE_PARALLEL)
  TARGET_LINK_LIBRARIES(WSTITK vtkParallelTCL)
ENDIF (VTK_USE_PARALLEL)

TARGET_LINK_LIBRARIES(WSTITK vtkWSTCommonTCL vtkWSTCommon vtkVTKITKCommonTCL vtkVTKITKCommon ITKNumerics)

INSTALL_TARGETS(/bin WSTITK)
